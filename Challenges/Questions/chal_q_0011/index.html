Consider the following macro definition:

#define min(X, Y) ((X) < (Y) ? (X) : (Y))

and the following four functions: 

  1. int get_sum(int x) {
    int sum = 0;
    
    for (int i = 0; i < x; i++) {
        sum += i;
    }
    
    return sum;
}


  2. int get_next_even_number(void) {
    static int number = -2;
    
    number += 2;
    
    return number;
}


  3. int get_sum_of_squares(int x) {
    int sum = 0;
    for (int i = 0; i < x; i++) {
        sum += i*i;
    }

    return sum;
}


  4. unsigned get_sum_safely(unsigned x, unsigned y) {
    if (x + y < x) {
        printf("Overflow in function get_sum_safely!");
        return 0;
    }

    return x + y;
}

Which of these statements can cause something unintended to happen? (in C)

Assume get_sum, get_next_even_number and get_sum_of_squares do not overflow in the below statements.


Statement 1 
result = min(10, get_sum(x));

Statement 2 
result = min(10, get_next_even_number());

Statement 3 
result = min(10, get_sum_of_squares(y));

Statement 4
